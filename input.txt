xi_init = 0.0;
xf_init = 2.0;
xi = xi_init;
xf = xf_init;
epsilon = 0.00000000000001;
f = @(x) (-(x*x) + 2.0);
x = (xi+xf)/2;
while(((xf-xi)^ 2) ^ 0.5 > epsilon)
  if( (f(xf) > 0) & (f(xi) < 0))
    if(f(x) > 0)
      xf = x;
    else
      xi = x;
    end;
  else
    if(f(x) < 0)
      xf = x;
    else
      xi = x;
    end;
  end;
  x = (xi+xf)/2;
end;
