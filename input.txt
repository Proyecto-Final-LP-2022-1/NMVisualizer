xi_init = 0.5;
xf_init = 6.5;
xi = xi_init;
xf = xf_init;
epsilon = 0.000001;
f = @(x) sin(x);
x = (xi+xf)/2;
while(((xf-xi)^ 2) ^ 0.5 > epsilon)
  if( (f(xf) > 0) & (f(xi) < 0))
    if(f(x) > 0)
      xf = x;
    else
      xi = x;
    end;
  else
    if(f(x) < 0)
      xf = x;
    else
      xi = x;
    end;
  end;
  x = (xi+xf)/2;
end;
